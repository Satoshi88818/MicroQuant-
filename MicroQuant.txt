N_TUBULINS = 8                    # Small lattice for simulation speed (real ~10^4–10^6)

class TubulinQubit:
    """Single tubulin dimer as a quantum bit (two conformational/dipole states)."""
    def __init__(self, index: int):
        self.index = index
        # Basis: |0> = GDP-like (low dipole), |1> = GTP-like (high dipole)
        self.state = qt.basis(2, 0)  # Start in ground state

    def apply_dipole_hamiltonian(self, coupling_strength: float = 1.0):
        """Local Hamiltonian for pi-electron dipole oscillation."""
        # σ_z term for dipole energy + small σ_x for tunneling
        return coupling_strength * (0.5 * qt.sigmaz() + 0.1 * qt.sigmax())


class MicrotubuleLattice:
    """Full microtubule as a 1D chain of coupled tubulins (scalable to 2D lattice)."""
    
    def __init__(self, n_tubulins: int = N_TUBULINS):
        self.n = n_tubulins
        self.tubulins = [TubulinQubit(i) for i in range(n)]
        self.hamiltonian = self._build_hamiltonian()
        self.current_state = qt.tensor([t.state for t in self.tubulins])
        self.time_series = []          # For animation / metrics
        self.coherence_history = []
        self.entanglement_history = []

    def _build_hamiltonian(self) -> qt.Qobj:
        """Dipole-dipole interaction Hamiltonian (XY model + local terms)."""
        H = qt.tensor([qt.qeye(2) for _ in range(self.n)]) * 0.0
        # Local dipole energy
        for i in range(self.n):
            H += qt.tensor([qt.qeye(2)] * i + [self.tubulins[i].apply_dipole_hamiltonian()] +
                           [qt.qeye(2)] * (self.n - i - 1))
        # Nearest-neighbor coupling (superradiant energy transfer)
        J = 2.0  # Coupling strength (arbitrary units)
        for i in range(self.n - 1):
            sigma_plus_i = qt.tensor([qt.qeye(2)] * i + [qt.sigmap()] + [qt.qeye(2)] * (self.n - i - 1))
            sigma_minus_j = qt.tensor([qt.qeye(2)] * (i + 1) + [qt.sigmam()] + [qt.qeye(2)] * (self.n - i - 2))
            H += J * (sigma_plus_i * sigma_minus_j.dag() + sigma_plus_i.dag() * sigma_minus_j)
        return H

    def evolve_coherent(self, t_list: np.ndarray) -> None:
        """Unitary evolution under the microtubule Hamiltonian (normal state)."""
        result = qt.mesolve(self.hamiltonian, self.current_state, t_list, c_ops=[])
        self.current_state = result.states[-1]
        self.time_series.append(t_list[-1])
        
        # Record metrics
        self.coherence_history.append(self._measure_coherence())
        self.entanglement_history.append(self._measure_entanglement())

    def apply_anesthetic(self, strength: float = ANESTHETIC_DECAY):
        """Apply Lindblad decoherence operators (anesthetic docking on pi-clouds)."""
        c_ops = []
        for i in range(self.n):
            # Local dephasing + amplitude damping on each tubulin
            proj = qt.tensor([qt.qeye(2)] * i + [qt.sigmam()] + [qt.qeye(2)] * (self.n - i - 1))
            c_ops.append(np.sqrt(strength) * proj)
        # Collapse to mixed state via master equation (one step)
        result = qt.mesolve(self.hamiltonian, self.current_state, [0, DT], c_ops=c_ops)
        self.current_state = result.states[-1]

    def _measure_coherence(self) -> float:
        """Off-diagonal density matrix elements (global coherence metric)."""
        rho = self.current_state * self.current_state.dag()
        off_diag = np.abs(rho.full()[0, -1])  # Example: |0...0><1...1| element
        return float(off_diag)

    def _measure_entanglement(self) -> float:
        """Simplified von Neumann entropy of half-chain bipartition."""
        rho = self.current_state * self.current_state.dag()
        half = self.n // 2
        rho_A = qt.ptrace(rho, list(range(half)))
        evals = rho_A.eigenenergies()
        evals = evals[evals > 1e-12]
        entropy = -np.sum(evals * np.log2(evals))
        return float(entropy)

    def perform_orchestrated_objective_reduction(self) -> Dict:

            # Collapse to one classical basis state (random but weighted)
            probs = np.abs(self.current_state.full().flatten()) ** 2
            outcome_idx = np.random.choice(range(2**self.n), p=probs)
            collapsed = qt.basis(2**self.n, outcome_idx)
            self.current_state = collapsed
            return {"collapsed": True, "outcome": bin(outcome_idx)[2:].zfill(self.n), "energy": energy}
        return {"collapsed": False, "outcome": None, "energy": energy}


# ========================== CLASSICAL FRÖHLICH BACKUP ==========================
def frohlich_oscillator_model(y, t, N, omega, pump):
    """Classical coupled oscillators simulating Fröhlich condensate pumping."""
    dydt = np.zeros(2 * N)
    for i in range(N):
        # Position and velocity
        x = y[2 * i]
        v = y[2 * i + 1]
        force = -omega[i]**2 * x
        # Nearest-neighbor coupling
        if i > 0:
            force += 0.5 * (y[2 * (i - 1)] - x)
        if i < N - 1:
            force += 0.5 * (y[2 * (i + 1)] - x)
        # Metabolic pumping (non-linear term approximated)
        force += pump * (1 - x**2) * v
        dydt[2 * i] = v
        dydt[2 * i + 1] = force
    return dydt
